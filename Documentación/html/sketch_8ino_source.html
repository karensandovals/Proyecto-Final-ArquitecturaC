283void mostrarMenu() {

286 int scrollDelay = 1000;

288 unsigned long lastScrollTime = 0;
289
290 while (key != 'D') {
291 if (millis() - lastScrollTime >= scrollDelay) {
292 lcd.clear();
293 lcd.setCursor(0, 0);
294 for (int i = 0; i < 2; i++) {
295 int currentOption = opcion + i;
296 lcd.setCursor(0, i);
297 if (currentOption == 0) {
298 lcd.print("1. UmbralTempHigh");
299 } else if (currentOption == 1) {
300 lcd.print("2. UmbralTempLow ");
301 } else if (currentOption == 2) {
302 lcd.print("3. UmbralLuzHigh ");
303 } else if (currentOption == 3) {
304 lcd.print("4. UmbralLuzLow ");
305 } else if (currentOption == 4) {
306 lcd.print("5. Monitoreo");
307 } else if (currentOption == 5) {
308 lcd.print("6. Reset");
309 }
310 }
311
312 lastScrollTime = millis();
313 }
314
315 key = keypad.getKey();
316
317 if (key == 'A') {
318 opcion = (opcion - 1 + 6) % 6;
319 } else if (key == 'B') {
320 opcion = (opcion + 1) % 6;
321 } else if (key >= '1' && key <= '6') {
322 ejecutarOpcion(key);
323 key = 'D';
324 }
325 }
